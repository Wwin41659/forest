Struct tabled::settings::style::BorderText

pub struct BorderText<L> { /* private fields */ }
Expand description

BorderText writes a custom text on a border.

Example

use tabled::{Table, settings::style::BorderText, settings::object::Rows};

let mut table = Table::new(["Hello World"]);
table
    .with(BorderText::new("+-.table").horizontal(Rows::first()));

assert_eq!(
    table.to_string(),
    "+-.table------+\n\
     | &str        |\n\
     +-------------+\n\
     | Hello World |\n\
     +-------------+"
);
